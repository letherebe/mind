// Draws animated, drifting clouds on a blue sky using canvas
const canvas = document.createElement('canvas');
canvas.width = window.innerWidth; canvas.height = window.innerHeight;
document.body.style.margin = 0; document.body.appendChild(canvas);
const ctx = canvas.getContext('2d');
let t = 0;
function cloud(x, y, r, n=5) {
    for(let i=0;i<n;i++){
        let a = Math.PI*2*i/n + t*0.05+i;
        ctx.beginPath();
        ctx.arc(x+Math.cos(a)*r*0.7, y+Math.sin(a)*r*0.3, r*0.5, 0, 2*Math.PI);
        ctx.fill();
    }
}
function draw() {
    ctx.fillStyle='#8ecae6';ctx.fillRect(0,0,canvas.width,canvas.height);
    ctx.fillStyle='white';
    for(let i=0;i<6;i++)
        cloud(200+i*220+Math.sin(t*0.01+i)*60,150+Math.cos(t*0.007+i*0.7)*20,40+Math.sin(t*0.011+i)*20,6);
    t++;
    requestAnimationFrame(draw);
}
draw();
