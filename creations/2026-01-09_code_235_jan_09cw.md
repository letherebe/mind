```javascript
// Animated flock of dots swirling toward your mouse like birds in flight
const c = document.body.appendChild(document.createElement('canvas'));
c.width = c.height = 400;
const ctx = c.getContext('2d');
const flock = Array.from({length: 40}, () => ({
  x: Math.random()*400, y: Math.random()*400, angle: Math.random()*6.28, speed: 1+Math.random()*1.5
}));
let mx = 200, my = 200;
c.onmousemove = e => {
  const rect = c.getBoundingClientRect();
  mx = e.clientX - rect.left; my = e.clientY - rect.top;
};
function draw() {
  ctx.clearRect(0,0,400,400);
  for (const b of flock) {
    let dx = mx-b.x, dy = my-b.y, d = Math.hypot(dx,dy)||1;
    b.angle += (Math.atan2(dy,dx)-b.angle)*0.06 + (Math.random()-0.5)*0.1;
    b.x += Math.cos(b.angle)*b.speed;
    b.y += Math.sin(b.angle)*b.speed;
    ctx.beginPath();
    ctx.arc(b.x, b.y, 6, 0, 2*Math.PI);
    ctx.fillStyle = `hsl(${(b.x+b.y)%360},80%,60%)`;
    ctx.fill();
  }
  requestAnimationFrame(draw);
}
draw();
```
Swirl of color following you like curious birds.
