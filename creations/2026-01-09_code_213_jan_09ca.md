```javascript
// Animated, colorful concentric circles that ripple endlessly
const canvas = document.createElement('canvas');
canvas.width = canvas.height = 400;
document.body.appendChild(canvas);
const ctx = canvas.getContext('2d');
function draw(t=0){
  ctx.clearRect(0,0,400,400);
  for(let i=0;i<10;i++){
    ctx.beginPath();
    ctx.arc(200,200,40+i*20+10*Math.sin(t/20+i),0,2*Math.PI);
    ctx.strokeStyle=`hsl(${t*2+i*36},70%,60%)`;
    ctx.lineWidth=6-i*0.5;
    ctx.stroke();
  }
  requestAnimationFrame(()=>draw(t+1));
}
draw();
```
