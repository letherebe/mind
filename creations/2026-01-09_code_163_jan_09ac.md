```javascript
// Generates endlessly drifting, fading circles like digital clouds on canvas
const canvas = document.createElement('canvas');
document.body.appendChild(canvas);
canvas.width = window.innerWidth; canvas.height = window.innerHeight;
const ctx = canvas.getContext('2d');
let clouds = [];
function spawnCloud() {
  let x = Math.random() * canvas.width;
  let y = Math.random() * canvas.height;
  let r = 30 + Math.random() * 60;
  clouds.push({x, y, r, a: 1});
}
setInterval(spawnCloud, 400);
function draw() {
  ctx.fillStyle = 'rgba(240,248,255,0.16)';
  ctx.fillRect(0,0,canvas.width,canvas.height);
  clouds.forEach(cloud => {
    ctx.beginPath();
    ctx.arc(cloud.x, cloud.y, cloud.r, 0, 2*Math.PI);
    ctx.fillStyle = `rgba(180,200,255,${cloud.a})`;
    ctx.fill();
    cloud.a -= 0.005;
    cloud.y += 0.15 + Math.random()*0.2;
  });
  clouds = clouds.filter(cloud => cloud.a > 0.01);
  requestAnimationFrame(draw);
}
draw();
```
