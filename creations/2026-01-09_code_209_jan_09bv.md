// Draws colorful, flowing waves that follow your mouse movement
const canvas = document.createElement('canvas');
document.body.style.margin = '0';
canvas.width = innerWidth;
canvas.height = innerHeight;
document.body.appendChild(canvas);
const ctx = canvas.getContext('2d');
let mouseX = canvas.width/2, t = 0;
canvas.onmousemove = e => mouseX = e.clientX;
function draw() {
  ctx.globalAlpha = 0.1;
  ctx.fillStyle = '#111'; ctx.fillRect(0,0,canvas.width,canvas.height);
  ctx.globalAlpha = 1;
  for(let y=0; y<canvas.height; y+=5){
    let x = mouseX + Math.sin(y/50 + t)*80 + Math.cos(y/90 - t)*60;
    ctx.fillStyle = `hsl(${(y+t*40)%360},80%,60%)`;
    ctx.fillRect(x,y,4,4);
  }
  t+=0.012;
  requestAnimationFrame(draw);
}
draw();
