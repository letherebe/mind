```javascript
// Animated spiral of dots expanding and contracting like a breathing galaxy
const c=document.body.appendChild(document.createElement('canvas'));
c.width=window.innerWidth;c.height=window.innerHeight;
const x=c.width/2,y=c.height/2,ctx=c.getContext('2d');
let t=0;
function draw(){
  ctx.fillStyle='#000a18';ctx.fillRect(0,0,c.width,c.height);
  for(let i=0;i<200;i++){
    let a=i*0.2+t*0.015,r=80+60*Math.sin(t*0.02+i*0.025);
    ctx.beginPath();
    ctx.arc(x+Math.cos(a)*r,y+Math.sin(a)*r,2,0,7);
    ctx.fillStyle=`hsl(${i*2+t*2},70%,60%)`;
    ctx.fill();
  }
  t++;
  requestAnimationFrame(draw);
}
draw();
```
